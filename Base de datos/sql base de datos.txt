CREATE EXTENSION IF NOT EXISTS "pgcrypto";

-- Limpiar tablas anteriores (orden correcto por FK)
DROP TABLE IF EXISTS resultados_test CASCADE;
DROP TABLE IF EXISTS categorias_resultado CASCADE;

-- =============================================
-- TABLA 1: categorias_resultado  ← se crea PRIMERO
-- =============================================
CREATE TABLE categorias_resultado (
    id           INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    nombre       VARCHAR(20)   NOT NULL,
    promedio_min NUMERIC(3,2)  NOT NULL,
    promedio_max NUMERIC(3,2)  NOT NULL
);

CREATE INDEX idx_categorias_nombre ON categorias_resultado (nombre);
CREATE INDEX idx_categorias_rango  ON categorias_resultado (promedio_min, promedio_max);

COMMENT ON TABLE  categorias_resultado              IS 'Categorías del test emocional Illari';
COMMENT ON COLUMN categorias_resultado.nombre       IS 'Nombre de la categoría (ej: "Alto")';
COMMENT ON COLUMN categorias_resultado.promedio_min IS 'Límite inferior inclusivo del rango';
COMMENT ON COLUMN categorias_resultado.promedio_max IS 'Límite superior inclusivo del rango';

-- Categorías base (el backend las usa para asignar automáticamente)
INSERT INTO categorias_resultado (nombre, promedio_min, promedio_max) VALUES
    ('AP',    0.00, 2.99),   -- Atención Profesional  (avg < 3)
    ('SA',    3.00, 3.99),   -- Apoyo Profesional     (avg 3-3.99)
    ('SF',    4.00, 5.00);   -- Felicitación          (avg >= 4)


-- =============================================
-- TABLA 2: resultados_test  ← se crea DESPUÉS
-- =============================================
CREATE TABLE resultados_test (
    id             SERIAL PRIMARY KEY,
    usuario_id     UUID          DEFAULT gen_random_uuid() NOT NULL,
    promedio       NUMERIC(3,2)  NOT NULL,
    categoria_id   INTEGER       NOT NULL
                   REFERENCES categorias_resultado(id) ON DELETE RESTRICT,
    nivel          VARCHAR(50)   NOT NULL,
    color          VARCHAR(20)   NOT NULL,
    mensaje        TEXT          NOT NULL,
    fecha_registro TIMESTAMP WITHOUT TIME ZONE DEFAULT CURRENT_TIMESTAMP NOT NULL
);

CREATE INDEX idx_resultados_fecha     ON resultados_test (fecha_registro DESC);
CREATE INDEX idx_resultados_categoria ON resultados_test (categoria_id);

COMMENT ON TABLE  resultados_test              IS 'Resultados anónimos del test emocional Illari';
COMMENT ON COLUMN resultados_test.usuario_id   IS 'UUID anónimo generado por sesión';
COMMENT ON COLUMN resultados_test.promedio     IS 'Promedio calculado del test';
COMMENT ON COLUMN resultados_test.categoria_id IS 'FK — categoría asignada según el promedio';
COMMENT ON COLUMN resultados_test.nivel        IS 'Nivel: AP, SA o SF';
COMMENT ON COLUMN resultados_test.color        IS 'Color para la UI: red, yellow o green';
COMMENT ON COLUMN resultados_test.mensaje      IS 'Mensaje personalizado del resultado';


-- =============================================
-- VISTA: consultas fáciles en pgAdmin
-- =============================================
CREATE OR REPLACE VIEW v_resultados AS
SELECT
    rt.id,
    rt.usuario_id,
    rt.promedio,
    cr.nombre        AS categoria,
    rt.nivel,
    rt.color,
    rt.fecha_registro
FROM resultados_test rt
JOIN categorias_resultado cr ON cr.id = rt.categoria_id
ORDER BY rt.fecha_registro DESC;
